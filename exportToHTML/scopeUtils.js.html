function isReferenceToGlobalFunction(calleeName, node, sourceCode) {
    const ref = sourceCode
        .getScope(node)
        .references.find(ref => ref.identifier.name === calleeName);
    // ensure it's the "global" version
    return !ref?.resolved?.defs.length;
}
