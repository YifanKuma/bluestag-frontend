const reRegExpChar = /[\\^$.*+?()[\]{}|]/g;
const reHasRegExpChar = RegExp(reRegExpChar.source);
function escapeRegExp(string = '') {
    return string && reHasRegExpChar.test(string)
        ? string.replaceAll(reRegExpChar, '\\$&')
        : string;
