/** @type {(str: string) => string} */
function stripBOM(str) {
  return str.replace(/^\uFEFF/, '');
}

/** @type {import('./readPkgUp').default} */
exports.default = function readPkgUp(opts) {
  const fp = pkgUp(opts);

  if (!fp) {
    return {};
  }

  try {
    return {
      pkg: JSON.parse(stripBOM(fs.readFileSync(fp, { encoding: 'utf-8' }))),
      path: fp,
    };
  } catch (e) {
    return {};
  }
};
